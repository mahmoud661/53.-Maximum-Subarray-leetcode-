class Solution(object):
    def maxSubArray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        maxo = nums[0]
        sum = 0
        for i in range(len(nums)):
            sum +=nums[i]
            maxo = max(maxo,sum)
            if sum<0:
                sum = 0

        return maxo
